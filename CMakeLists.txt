cmake_minimum_required(VERSION 3.15)
project(TetrisGame)

set(CMAKE_CXX_STANDARD 20)

# 모든 소스 파일 재귀적으로 포함
file(GLOB_RECURSE SOURCES CONFIGURE_DEPENDS
    "blocks/**/*.cpp"
    "board/*.cpp"
    "core/*.cpp"
    "domain/*.cpp"
    "presentation/*.cpp"
    "utils/*.cpp"
    "main.cpp"
)

add_executable(TetrisGame ${SOURCES})
